
#include <iostream>
using namespace std;
int ReadNumber()
{
	int Number;
	cout << "\n please Enter a number?";
	cin >> Number;
	return Number;
}
void AddArrayElement(int Number,int arr[100], int& arrlength)
{
	arrlength++;
	arr[arrlength - 1] = Number;
}
void InuputUserNumberInArray(int arr[100], int& arrlength)
{
	bool AddMore = true;
	do
	{
		AddArrayElement(ReadNumber(), arr, arrlength);
		cout << "\n Do you want add more number?[0]:No,[1]:Yes?";
		cin >> AddMore;
	} while (AddMore);
}
void PrintArray(int arr[100], int arrlength)
{
	for (int i = 0; i < arrlength; i++)
		cout << arr[i] << " ";
	cout << "\n";
}
int main()
{
	int arr[100], arrlength = 0;
	InuputUserNumberInArray(arr, arrlength);

	cout << "\nArray length : " << arrlength << endl;
	cout << "Array elements: ";
	PrintArray(arr, arrlength);
	return 0;
}
