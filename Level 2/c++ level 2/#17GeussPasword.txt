
#include <iostream>
#include <string>
using namespace std;
string ReadPassword()
{
	string password;
	cout << "please enter a 3-letter password (all capital)?\n";
	cin >> password;
	return password;
}
bool GuessPassword(string originalpassword)
{
	string word = "";
	int counter = 0;
	cout << "\n";
	for (int i = 65; i <= 90; i++)
	{
		for (int j = 65; j <= 90; j++)
		{
			for (int k = 65; k <= 90; k++)
			{
				word = word + char(i);
				word = word + char(j);
				word = word + char(k);
				counter++;
				cout << "Trial[" << counter << "]:";
				cout << word << endl;

				if (word == originalpassword) {
					cout << "\n password is:" << word << "\n";
					cout << "found after ";
					cout << counter << " Trial(s)\n";
					return true;
				}
				word = "";
			}
		}
	}
	return false;
}
int main()
{
	GuessPassword(ReadPassword());
	return 0;
}
