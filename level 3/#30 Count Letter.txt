#include<iostream>
#include<string>
using namespace std;

string ReadString()
{
	string S1;
	cout << "please Enter your string ?:\n";
	getline(cin, S1);
	return S1;
}
char ReadChar()
{
	char Ch1;
	cout << "\n please Enter a Character?\n";
	cin >> Ch1;
	return Ch1;
}

short CountLetter(string S1, char letter)
{
	short Counter = 0;
	for (short i = 0; i < S1.length(); i ++ )
	{
		if (S1[i] == letter)
			Counter++;
	}
	return Counter;
}

int main()
{
	string S1 = ReadString();
	char Ch1 = ReadChar();

	cout << "\n letter\'" << Ch1 << "\'count=" << CountLetter(S1, Ch1);

	system("pause>0");
}
