#include<iostream>
#include<string>
#include<iomanip>
using namespace std;

int RandNumber(int From, int To)
{
	int randNum = rand() % (To - From + 1) + From;
	return randNum;
}


void FillArrWithRandomNumber(int arr[3][3], int Rows, int Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			arr[i][j] = RandNumber(1, 100);
		}
	}
}

void PrintMatrix(int arr[3][3], short Rows, short Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			cout << setw(4) << arr[i][j] << "  ";
		}
		cout << "\n";
	}
}

int Sumcols(int arr[3][3], short Rows, short ColNumber)
{
	int sum = 0;
	for (short i = 0; i <= Rows - 1; i++)
	{
		sum += arr[i][ColNumber];
	}
	return sum;
}


void PrintEachCols(int arr[3][3], short Rows, short Cols)
{
	cout << "\n the folloeinf are the sum of each row in the matrix:\n";
	for (short j = 0; j < Cols; j++)
	{
		cout << "Col" << j+ 1 << "Sum: " << Sumcols(arr,Rows, j) << endl;;
	}

}


int main()
{
	srand((unsigned)time(NULL));

	int arr[3][3];
	FillArrWithRandomNumber(arr, 3, 3);

	cout << "\n the following is 3x3 Matrix: \n";

	PrintMatrix(arr, 3, 3);

	PrintEachCols(arr, 3, 3);


	system("pause>0");

}
