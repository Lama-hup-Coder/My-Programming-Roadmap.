#include<iostream>
#include<string>
using namespace std;

string ReadString()
{
	string S1;
	cout << "please Enter your string ?:\n";
	getline(cin, S1);
	return S1;
}
char ReadChar()
{
	char Ch1;
	cout << "\n please Enter a Character?\n";
	cin >> Ch1;
	return Ch1;
}
char InvertLetterCase(char Ch1)
{
	return isupper(Ch1) ? tolower(Ch1) : toupper(Ch1);
}

short CountLetter(string S1, char letter,bool Matchcase=true)
{
	short Counter = 0;
	for (short i = 0; i < S1.length(); i++)
	{
		if (Matchcase)
		{
			if (S1[i] == letter)
				Counter++;
		}
		else
		{
			if (tolower(S1[i]) == tolower(letter))
				Counter++;
		}
	}
	return Counter;
}

int main()
{
	string S1 = ReadString();
	char Ch1 = ReadChar();

	cout << "\n letter\'" << Ch1 << "\'count=" << CountLetter(S1, Ch1);

	cout << "\n letter\'" << Ch1 << "\'";
	cout << "Or \'" << InvertLetterCase(Ch1) << "\'";
	cout << "count=" << CountLetter(S1, Ch1, false);

	system("pause>0");
}
