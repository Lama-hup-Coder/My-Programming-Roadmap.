#include<iostream>
using namespace std;

void PrintFibonacciUsingLoop(short Number)
{
	int FibNumber = 0;
	int prev2 = 0, prev1 = 1;

	cout << "1  ";
	for (short i = 2; i <= Number; ++i)
	{
		FibNumber = prev1 + prev2;
		cout << FibNumber << "  ";
		prev2 = prev1;
		prev1 = FibNumber;
	}
}


int main()
{
	PrintFibonacciUsingLoop(10);

	system("pause>0");
}
