#include<string>
#include<iostream>
using namespace std;

string ReadString()
{
	string s1;
	cout << "\n please enter the string?:";
	getline(cin,s1);
	return s1;
}
void PrintEachWordInString(string S1)
{
	string delim = " ";
	cout << "\n your string words are:\n\n";
	short pos = 0;
	string sWord;
	while ((pos = S1.find(delim)) != std::string::npos)
	{
		sWord = S1.substr(0, pos);
		if (sWord != "")
		{
			cout << sWord << endl;
		}
		S1.erase(0, pos + delim.length());
	}
	if (S1 != "")
	{
		cout << S1 << endl;
	}
	}
	

int main()
{
	PrintEachWordInString(ReadString());

	system("pause>0");
}
