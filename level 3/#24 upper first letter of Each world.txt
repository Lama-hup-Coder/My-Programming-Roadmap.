string UpperFirstLetterOfEachWorld(string S1)
{
	bool FirstLetter = true;
	for (short i = 0; i < S1.length(); i++)
	{
		if (S1[i] != ' ' && FirstLetter)
		{
			S1[i] = toupper(S1[i]);
		}
		FirstLetter = (S1[i] == ' ' ? true : false);
	}

	return S1;
}
