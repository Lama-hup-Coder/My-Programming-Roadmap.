#include<iostream>
#include<string>
using namespace std;

string ReadString()
{
	string S1;
	cout << "please Enter your string ?:\n";
	getline(cin, S1);
	return S1;
}

char InverLetterCase(char ch1)
{
	return isupper(ch1) ? tolower(ch1) : toupper(ch1);
}

string InvertAllStingLetters(string S1)
{
	for (short i = 0; i < S1.length(); i++)
	{
		S1[i] = InverLetterCase(S1[i]);
	}
	return S1;
}

int main()
{
	string S1 = ReadString();

	cout << "\n String after Inverting All Letter Case:\n";
	S1 = InvertAllStingLetters(S1);
	cout << S1 << endl;

	system("pause>0");

}
