#include<iostream>
#include<string>
#include<iomanip>
using namespace std;

void FillMatrixWithOrderedNumbers(int arr[3][3], short Rows, short Cols)
{
	short Counter = 0;
	for (short i = 0 ; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			Counter++;
			arr[i][j]=Counter;
		}
	}
}

void PrintMatrix(int arr[3][3], short Rows, short Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			cout << setw(4) << arr[i][j] << "  ";
		}
		cout << "\n";
	}
}

void PrintTrasposeMatrix(int arr[3][3], short Rows, short Cols)
{
	cout << "\n the folloein is a 3x3 Traspose Matrix:\n";
	for (short j = 0; j < Cols; j++)
	{
		for (short i = 0; i < Rows; i++)
		{
			cout<<setw(2) << arr[i][j] << " ";
		}
		cout << endl;
	}
}

int main()
{
	int arr[3][3];
	FillMatrixWithOrderedNumbers(arr, 3, 3);

	cout << "\n the folloein is a 3x3 Ordered Matrix:\n";
	PrintMatrix(arr, 3, 3);
	
	PrintTrasposeMatrix(arr, 3, 3);

	system("pause>0");

}
